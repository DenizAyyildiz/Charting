WITH PRDDATA AS
(SELECT C.WORKCENTER,C.PERSONELNUM,SUM(C.OUTPUT) AS QUANTITY,FORMAT(C.WORKEND, 'dd/MM/yyyy') AS DATE1

	FROM IASPRDCONF C
	LEFT OUTER JOIN IASPRDOPR P ON P.COMPANY = C.COMPANY
		AND P.CLIENT = C.CLIENT
		AND P.PLANT = C.PLANT
		AND C.CONFIRMATION = P.CONFIRMATION
	WHERE C.CLIENT = '00'
		AND C.COMPANY = '01'
		AND C.PLANT = '01'
		AND C.POTYPE = 'K'
		AND C.WORKEND < GETDATE()
		AND C.WORKEND >=  DATEADD(DAY,-700,GETDATE())
	GROUP BY C.WORKCENTER,C.PERSONELNUM,C.WORKEND
)
SELECT PRDDATA.WORKCENTER,PRDDATA.PERSONELNUM,PRDDATA.DATE1,SUM(PRDDATA.QUANTITY) AS QUANTITY,PRDDATA.DATE1 FROM PRDDATA
GROUP BY PRDDATA.DATE1, PRDDATA.WORKCENTER, PRDDATA.PERSONELNUM
