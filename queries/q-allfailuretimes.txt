SELECT P.MATERIAL, F.BREAKDOWNSTART AS WORKSTART, F.BREAKDOWNEND AS WORKEND,0.0 AS SESSIONTIME,

	 F.WORKCENTER,C.PERSONELNUM, IASPRD005X.STEXT AS FAILURETEXT

	FROM IASPRDFAILURE F , IASPRDOPR P , IASPRDCONF C,IASMATBASIC , IASPRD005
	LEFT OUTER JOIN IASPRD005X ON (IASPRD005X.CLIENT = IASPRD005.CLIENT
		AND IASPRD005X.COMPANY = IASPRD005.COMPANY
		AND IASPRD005X.PLANT = IASPRD005.PLANT
		AND IASPRD005.FAILURECODE = IASPRD005X.FAILURECODE
		AND IASPRD005X.LANGU = 'T')
	WHERE F.CLIENT = '00'
		AND F.COMPANY = '01'
		AND F.PLANT = '01'
		AND P.MATERIAL LIKE '%'
		AND F.BREAKDOWNSTART < GETDATE()
		AND F.BREAKDOWNSTART >= DATEADD(DAY,-720,GETDATE())
		AND IASMATBASIC.MATERIAL = P.MATERIAL
		AND IASMATBASIC.COMPANY = P.COMPANY
		AND IASPRD005.CLIENT = F.CLIENT
		AND IASPRD005.COMPANY = F.COMPANY
		AND IASPRD005.PLANT = F.PLANT
		AND P.CLIENT = F.CLIENT
		AND P.COMPANY = F.COMPANY
		AND P.PLANT = F.PLANT
		AND P.CONFIRMATION = F.CONFIRMATION
		AND P.PRDORDER = F.PRDORDER
		AND F.WORKCENTER <> ''
		AND IASPRD005.FAILURECODE = F.FAILURECODE
		AND F.POTYPE LIKE '%'
		AND P.POTYPE = F.POTYPE
		AND C.CONFIRMATION = P.CONFIRMATION
		AND C.CONFIRMPOS = P.CONFIRMPOS
		AND P.COMPANY = C.COMPANY
		AND P.CLIENT = C.CLIENT
		AND P.PLANT = C.PLANT
		AND P.POTYPE = C.POTYPE
