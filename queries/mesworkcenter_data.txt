SELECT  P.VLFSIRA,A.WORKCENTER,M.DRAWNUM AS MATERIAL,
ISNULL(T.PARTITION,0) AS PARTITION, ISNULL(R.NDEVIRHIZI,0) AS NDEVIRHIZI,
420* ISNULL(R.NDEVIRHIZI,0) AS KALAN ,
(ISNULL(P.BASEQUAN,0)- ISNULL(P.OUTPUT,0)) AS OPTIMALMIKTAR
 FROM IASAUTSTATUS A
LEFT JOIN IASPRDOPR P ON A.CONFIRMATION = P.CONFIRMATION AND A.CONFIRMPOS = P.CONFIRMPOS
LEFT JOIN IASPRDRST  T ON T.PRDORDER = P.PRDORDER AND T.OPERATION = P.OPERATION
LEFT JOIN IASROUOPR R ON P.OPERATION = R.OPERATION AND P.MATERIAL = R.MATERIAL
LEFT JOIN IASMATBASIC M ON M.MATERIAL = P.MATERIAL AND M.COMPANY = '01'
WHERE A.WORKCENTER IN
